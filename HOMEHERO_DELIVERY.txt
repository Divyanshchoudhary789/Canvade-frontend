───────────────────────────────────────────────────────────────────────────────
                         CUSTOM HOOKS REFERENCE
───────────────────────────────────────────────────────────────────────────────

1️⃣  useResponsiveDesign()
    ├─ Purpose: Track device type
    ├─ Returns: { width, isMobile, isTablet, isDesktop, isWide }
    ├─ Features: SSR safe, debounced, cleanup
    └─ Usage: const { isMobile } = useResponsiveDesign();

2️⃣  useFadeInAnimation()
    ├─ Purpose: Initial fade-in animation
    ├─ Returns: boolean (isVisible)
    ├─ Features: Smooth timing, no jank
    └─ Usage: const isVisible = useFadeInAnimation();

3️⃣  useBackgroundRotation(images)
    ├─ Purpose: Rotate background images
    ├─ Returns: { currentIndex, isFading }
    ├─ Features: Fade transitions, auto-cycle
    └─ Usage: const { currentIndex, isFading } = useBackgroundRotation(images);

4️⃣  useFormState(initialState)
    ├─ Purpose: Form state management
    ├─ Returns: { formData, errors, touched, handlers, reset }
    ├─ Features: Validation ready, touched tracking
    └─ Usage: const form = useFormState({ field: '' });

5️⃣  useScrollToTop()
    ├─ Purpose: Scroll to top button logic
    ├─ Returns: { showButton, scrollToTop }
    ├─ Features: Smart visibility, smooth scroll
    └─ Usage: const { showButton, scrollToTop } = useScrollToTop();
